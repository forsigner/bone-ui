# Input

## import

```jsx
import { Input, InputGroup, InputAddon, InputElement } from '@bone-ui/input'
```

## Basic

```jsx live
<Input placeholder="user name" />
```

## Variants

```jsx live
<Box spaceY3>
  <Input variant="outline" placeholder="outline" />
  <Input variant="filled" placeholder="filled" />
  <Input variant="unstyled" placeholder="unstyled" />
</Box>
```

## Color scheme

```jsx live
<Box spaceY3>
  <Input placeholder="user name" colorScheme="red400" />
  <Input placeholder="user name" colorScheme="blue400" />
  <Input placeholder="user name" colorScheme="purple400" />
</Box>
```

## Preset Sizes

```jsx live
<Box spaceY3>
  <Input size="sm" placeholder="small" />
  <Input size="md" placeholder="medium" />
  <Input size="lg" placeholder="large" />
</Box>
```

## Any Sizes

```jsx live
<Box spaceY3 w-200>
  <Input size={20} placeholder="size 20" />

```

## Rounded
